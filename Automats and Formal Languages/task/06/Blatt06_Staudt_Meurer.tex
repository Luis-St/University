	whilenot iszero(B) do # Dekrementiert B und C, bis B 0 ist
		pred(B);
		succ(C);
	od;
else # Wenn B 0 ist, dann ist A der Abstand
	whilenot iszero(A) do # Dekrementiert A und C, bis A 0 ist
